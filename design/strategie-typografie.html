7. CSS Variables (Zum Kopieren)

/* ============================================
   FONT IMPORTS
   ============================================ */
@import url('https://fonts.googleapis.com/css2?family=Open+Sans:wght@400;500;600;700&family=Roboto:wght@400;500;600;700&family=Roboto+Mono:wght@400;500;600&display=swap');

:root {
  /* ============================================
     FONT FAMILIES
     ============================================ */
  --font-heading: 'Open Sans', system-ui, sans-serif;
  --font-body: 'Roboto', system-ui, sans-serif;
  --font-mono: 'Roboto Mono', ui-monospace, monospace;
  
  /* ============================================
     FONT SIZES
     ============================================ */
  --text-display-lg: 3rem;      /* 48px */
  --text-display-md: 2.25rem;   /* 36px */
  --text-h1: 1.75rem;           /* 28px */
  --text-h2: 1.375rem;          /* 22px */
  --text-h3: 1.125rem;          /* 18px */
  --text-h4: 1rem;              /* 16px */
  --text-body-lg: 1rem;         /* 16px */
  --text-body-md: 0.875rem;     /* 14px */
  --text-body-sm: 0.8125rem;    /* 13px */
  --text-label-lg: 0.875rem;    /* 14px */
  --text-label-md: 0.75rem;     /* 12px */
  --text-label-sm: 0.6875rem;   /* 11px */
  --text-caption: 0.75rem;      /* 12px */
  --text-amount-lg: 1.5rem;     /* 24px */
  --text-amount-md: 1.125rem;   /* 18px */
  --text-amount-sm: 0.875rem;   /* 14px */
  
  /* ============================================
     FONT WEIGHTS
     ============================================ */
  --font-regular: 400;
  --font-medium: 500;
  --font-semibold: 600;
  --font-bold: 700;
  
  /* ============================================
     LINE HEIGHTS
     ============================================ */
  --leading-none: 1;
  --leading-tight: 1.25;
  --leading-snug: 1.375;
  --leading-normal: 1.5;
  --leading-relaxed: 1.625;
  --leading-loose: 1.75;
  
  /* ============================================
     LETTER SPACING
     ============================================ */
  --tracking-tighter: -0.05em;
  --tracking-tight: -0.025em;
  --tracking-normal: 0;
  --tracking-wide: 0.025em;
  --tracking-wider: 0.05em;
  --tracking-widest: 0.1em;
}
